Example

Here's a simple example of how to use My Project:

#include "matr.h"
int main() {
const size_t rows = 8000;
const int cols = 8000;
Matrix<float> A(rows, cols);
Matrix<float> B(rows, cols);
for (float i = 0; i < rows; ++i) {
for (float j = 0; j < cols; ++j) {
A(i, j) = i+j;
B(i, j) = j-j;
}
}
Matrix<float> D = A * B; // 重写*
}
